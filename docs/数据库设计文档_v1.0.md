## 一、数据库概览

使用 **Supabase (PostgreSQL)** 作为后端数据库。

### 核心表

| 表名 | 用途 |
|-----|------|
| `artists` | 艺人/团体信息 |
| `events` | 活动（演唱会/签售/快闪等） |
| `news` | 新闻资讯 |
| `flash_news` | AI生成的快讯 |
| `users` | 用户信息 |
| `favorites` | 用户收藏 |
| `agent_logs` | Agent执行日志 |

## 二、详细表结构

### 2.1 artists（艺人表）

```sql
CREATE TABLE artists (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  
  -- 名称（多语言）
  name_ko VARCHAR(100),        -- 韩文名
  name_zh VARCHAR(100),        -- 中文名  
  name_en VARCHAR(100),        -- 英文名
  
  -- 基本信息
  group_name VARCHAR(100),     -- 团体名（Solo艺人则为空）
  agency VARCHAR(100),         -- 经纪公司（SM/YG/HYBE/JYP等）
  country VARCHAR(50),         -- 所属国（韩国/中国等）
  debut_date DATE,             -- 出道日期
  
  -- 图像
  profile_image TEXT,          -- 头像URL（400x400）
  banner_image TEXT,           -- 横图URL（1920x600）
  
  -- 社交媒体
  social_links JSONB DEFAULT '{}',  -- {weverse: "", twitter: "", instagram: "", youtube: "", tiktok: ""}
  
  -- 属性
  artist_type VARCHAR(20),    -- group, solo, duo
  member_count INT,            -- 成员数（团体用）
  tags TEXT[],               -- 标签 [流行, Kpop, 女团, 四代]
  
  -- 状态
  status VARCHAR(20) DEFAULT 'active', -- active, hiatus, disbanded, solo_debut
  hot_score INT DEFAULT 0,     -- 热度指数（每日更新）
  
  -- SEO
  slug VARCHAR(100) UNIQUE,    -- URL友好的slug
  
  -- 时间戳
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_artists_slug ON artists(slug);
CREATE INDEX idx_artists_hot ON artists(hot_score DESC);
CREATE INDEX idx_artists_agency ON artists(agency);
```

### 2.2 events（活动表）

```sql
CREATE TABLE events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  
  -- 基本信息
  title VARCHAR(200),         -- 活动标题
  subtitle VARCHAR(200),       -- 副标题（如巡演名称）
  
  -- 关联
  artist_id UUID REFERENCES artists(id) ON DELETE SET NULL,
  
  -- 活动类型
  event_type VARCHAR(30),     -- concert, fansign, pop_up, fanmeeting, festival, broadcast, award
  
  -- 地点
  country VARCHAR(50),
  city VARCHAR(100),
  venue VARCHAR(200),
  address TEXT,
  latitude DECIMAL(10, 8),
  longitude DECIMAL(11, 8),
  
  -- 时间
  event_date TIMESTAMPTZ,     -- 活动开始时间
  event_end_date TIMESTAMPTZ,  -- 活动结束时间（多日活动）
  door_time TIMESTAMPTZ,      -- 入场时间
  
  -- 票务
  ticket_open_date TIMESTAMPTZ, -- 开票时间
  ticket_close_date TIMESTAMPTZ, -- 截票时间
  price_range VARCHAR(100),   -- "50000-150000"
  currency VARCHAR(10),       -- KRW, CNY, USD, JPY
  
  -- 链接
  official_link TEXT,         -- 官方链接
  ticket_link TEXT,           -- 票务链接
  poster_image TEXT,          -- 海报图
  
  -- 状态
  status VARCHAR(20) DEFAULT 'upcoming', -- upcoming, ticketing, sold_out, ended, cancelled
  is_highlighted BOOLEAN DEFAULT FALSE,  -- 是否重点推荐
  
  -- 热度
  hot_score INT DEFAULT 0,
  
  -- 来源
  source VARCHAR(50),        -- 抓取来源
  source_id VARCHAR(100),     -- 原始ID
  
  -- SEO
  slug VARCHAR(100) UNIQUE,
  
  -- 时间戳
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_events_date ON events(event_date);
CREATE INDEX idx_events_country ON events(country);
CREATE INDEX idx_events_artist ON events(artist_id);
CREATE INDEX idx_events_type ON events(event_type);
CREATE INDEX idx_events_status ON events(status);
CREATE INDEX idx_events_hot ON events(hot_score DESC);
```

### 2.3 news（新闻表）

```sql
CREATE TABLE news (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  
  -- 标题和内容
  title VARCHAR(300),         -- 中文标题（AI翻译后）
  summary TEXT,               -- AI生成的摘要（150字内）
  content TEXT,               -- 完整内容
  
  -- 原文
  original_title TEXT,        -- 原文标题
  original_content TEXT,      -- 原文内容
  language VARCHAR(10),       -- 原文语言 ko, zh, en, ja
  
  -- 来源
  source VARCHAR(100),       -- 来源媒体
  source_url TEXT,           -- 原文链接
  author VARCHAR(100),        -- 作者
  
  -- 关联
  artist_id UUID REFERENCES artists(id) ON DELETE SET NULL,
  
  -- 类型
  news_type VARCHAR(30),     -- comeback, event, achievement, controversy, schedule, interview
  tags TEXT[],
  
  -- 图像
  image_url TEXT,            -- 封面图
  images TEXT[],             -- 图集
  
  -- AI处理
  is_translated BOOLEAN DEFAULT FALSE,
  is_ai_summary BOOLEAN DEFAULT FALSE,
  
  -- 状态
  status VARCHAR(20) DEFAULT 'pending', -- pending, published, archived, featured
  priority INT DEFAULT 5,    -- 优先级 1-10
  
  -- 热度
  hot_score INT DEFAULT 0,
  view_count INT DEFAULT 0,
  
  -- SEO
  slug VARCHAR(100) UNIQUE,
  
  -- 时间戳
  publish_date TIMESTAMPTZ,  -- 发布时间
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_news_publish ON news(publish_date DESC);
CREATE INDEX idx_news_artist ON news(artist_id);
CREATE INDEX idx_news_type ON news(news_type);
CREATE INDEX idx_news_hot ON news(hot_score DESC);
```

### 2.4 flash_news（快讯表）

```sql
CREATE TABLE flash_news (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  
  -- 内容（AI生成）
  content TEXT NOT NULL,     -- 快讯内容，不超过100字
  
  -- 类型
  news_type VARCHAR(30),     -- comeback, announcement, ticketing, event, achievement
  priority INT DEFAULT 5,    -- 1-10
  
  -- 关联
  artist_id UUID REFERENCES artists(id) ON DELETE CASCADE,
  event_id UUID REFERENCES events(id) ON DELETE CASCADE,
  source_news_id UUID REFERENCES news(id) ON DELETE SET NULL,
  
  -- 状态
  is_active BOOLEAN DEFAULT TRUE,
  expires_at TIMESTAMPTZ,    -- 自动过期时间
  
  -- 时间戳
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_flash_active ON flash_news(is_active, expires_at);
```

### 2.5 users（用户表）

```sql
CREATE TABLE users (
  id UUID PRIMARY KEY REFERENCES auth.users(id),
  
  -- 个人信息
  nickname VARCHAR(50),
  avatar_url TEXT,
  bio TEXT,
  
  -- 设置

  country_filter VARCHAR(50)[],          -- 关注的国家筛选
  
  -- 统计
  following_count INT DEFAULT 0,
  
  -- 时间戳

  last_login TIMESTAMPTZ

```

### 2.6 favorites（收藏表）

```sql
CREATE TABLE favorites (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  
  -- 收藏类型（二选一）
  artist_id UUID REFERENCES artists(id) ON DELETE CASCADE,
  event_id UUID REFERENCES events(id) ON DELETE CASCADE,
  
  -- 提醒设置
  notify_ticket BOOLEAN DEFAULT TRUE,   -- 开票提醒
  notify_event BOOLEAN DEFAULT TRUE,    -- 活动开始提醒
  notify_news BOOLEAN DEFAULT FALSE,    -- 新闻更新提醒
  
  -- 时间戳
  created_at TIMESTAMPTZ DEFAULT NOW(),
  
  -- 唯一约束
  UNIQUE(user_id, artist_id),
  UNIQUE(user_id, event_id),
  
  CHECK (
    (artist_id IS NOT NULL AND event_id IS NULL) OR
    (artist_id IS NULL AND event_id IS NOT NULL)
  )
);
```

### 2.7 agent_logs（Agent日志表）

```sql
CREATE TABLE agent_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  
  agent_name VARCHAR(50),     -- agent_1_news_crawler
  task_type VARCHAR(30),     -- crawl, translate, parse, seo
  
  status VARCHAR(20),        -- running, success, failed, partial
  
  -- 输入输出
  input_data JSONB,
  output_data JSONB,
  error_message TEXT,
  
  -- 统计
  items_processed INT DEFAULT 0,
  items_created INT DEFAULT 0,
  items_updated INT DEFAULT 0,
  
  -- 时间
  started_at TIMESTAMPTZ,
  completed_at TIMESTAMPTZ,
  duration_ms INT,
  
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_agent_logs_name ON agent_logs(agent_name);
CREATE INDEX idx_agent_logs_status ON agent_logs(status);
CREATE INDEX idx_agent_logs_created ON agent_logs(created_at DESC);
```

## 三、RLS策略（行级安全）

### artists（公开可读）

```sql
-- 任何人可读
CREATE POLICY "Public read artists" ON artists
  FOR SELECT USING (status = 'active');

-- 认证用户可修改自己的收藏
CREATE POLICY "Users manage own favorites" ON favorites
  FOR ALL USING (auth.uid() = user_id);
```

### events（公开可读，Agent可写入）

```sql
-- 公开读
CREATE POLICY "Public read events" ON events
  FOR SELECT USING (status IN ('upcoming', 'ticketing', 'sold_out'));

-- Agent服务可写入（使用service role key）
```

## 四、常用查询

### 获取热门艺人（Top 20）

```sql
SELECT * FROM artists 
WHERE status = 'active' 
ORDER BY hot_score DESC 
LIMIT 20;
```

### 获取今日/近期活动

```sql
SELECT e.*, a.name_zh as artist_name, a.profile_image as artist_image
FROM events e
LEFT JOIN artists a ON e.artist_id = a.id
WHERE e.event_date >= NOW()
  AND e.status IN ('upcoming', 'ticketing')
ORDER BY e.event_date ASC
LIMIT 50;
```

### 获取艺人相关新闻

```sql
SELECT * FROM news 
WHERE artist_id = $1 
  AND status = 'published'
ORDER BY publish_date DESC 
LIMIT 10;
```

### 获取活跃快讯

```sql
SELECT fn.*, a.name_zh as artist_name
FROM flash_news fn
LEFT JOIN artists a ON fn.artist_id = a.id
WHERE fn.is_active = TRUE 
  AND (fn.expires_at IS NULL OR fn.expires_at > NOW())
ORDER BY fn.priority DESC, fn.created_at DESC
LIMIT 20;
```
